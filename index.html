# --- CONFIGURACIÓN INICIAL Y BIENVENIDA (GÉNESIS) ---
Function Show-Intro {
    Clear-Host
    Write-Host "=======================================================" -ForegroundColor 'DarkMagenta'

    Write-Host "`nIniciando la secuencia de optimización final..." -ForegroundColor 'Cyan'
    Write-Host "Este script desatará el verdadero potencial de tu sistema." -ForegroundColor 'Gray'
    Write-Host "ADVERTENCIA: Las optimizaciones son profundas. Procede con conocimiento." -ForegroundColor 'Red'
    Write-Host "======================================================="

# --- PASO 1: LIMPIEZA EXTREMA (LA GRAN PURGA) ---
Function Step1-The-Great-Purge {
    Write-Host "`n`n[PASO 1/11] - Ejecutando La Gran Purga..." -ForegroundColor 'Yellow'
    # Limpieza de archivos temporales
    Get-ChildItem -Path $env:temp, $env:windir\temp -Recurse -Force | Remove-Item -Recurse -Force -ErrorAction SilentlyContinue
    # Limpieza de caché de Windows Update
    Stop-Service wuauserv -Force
    Remove-Item -Path "$env:windir\SoftwareDistribution" -Recurse -Force -ErrorAction SilentlyContinue
    Start-Service wuauserv
    Write-Host "La Gran Purga ha concluido. El sistema es más ligero." -ForegroundColor Green
}

# --- PASO 2: OPTIMIZACIÓN DE MEMORIA (SINGULARIDAD DE MEMORIA) ---
Function Step2-Memory-Singularity {
    Write-Host "`n`n[PASO 2/11] - Alcanzando la Singularidad de Memoria..." -ForegroundColor 'Yellow'
    # Limpieza de caché de memoria en espera
    $Process = Get-WmiObject -Class Win32_OperatingSystem
    $Process.FreeAndNilSystemCache()
    Write-Host "La memoria ha sido condensada a su estado óptimo." -ForegroundColor Green
}

# --- PASO 3: INTEGRIDAD DEL NÚCLEO (REPARACIÓN CUÁNTICA) ---
Function Step3-Core-Integrity {
    Write-Host "`n`n[PASO 3/11] - Verificando la Integridad del Núcleo..." -ForegroundColor 'Yellow'
    sfc /scannow
    DISM /Online /Cleanup-Image /RestoreHealth
    Write-Host "El núcleo del sistema ha sido estabilizado y reforzado." -ForegroundColor Green
}

# --- PASO 4: ACTUALIZACIÓN DE POLÍTICAS (ACTUALIZADOR TAQUIÓNICO) ---
Function Step4-Tachyon-Updater {
    Write-Host "`n`n[PASO 4/11] - Aplicando Actualizador Taquiónico de Políticas..." -ForegroundColor 'Yellow'
    gpupdate /force
    Write-Host "Las políticas del sistema viajan ahora a la velocidad de la luz." -ForegroundColor Green
}

# --- PASO 5: REGISTRO - INTERFAZ (UI CÓSMICA) ---
Function Step5-Registry-Cosmic-UI {
    Write-Host "`n`n[PASO 5/11] - Configurando la Interfaz Cósmica del Registro..." -ForegroundColor 'Yellow'
    # Habilitar menú contextual extendido
    Set-ItemProperty -Path "HKCU:\Software\Microsoft\Windows\CurrentVersion\Explorer\Advanced" -Name "ExtendedUIHoverTime" -Value "0" -Type DWord -Force
    # Desactivar desenfoque en pantalla de bloqueo
    Set-ItemProperty -Path "HKLM:\SOFTWARE\Policies\Microsoft\Windows\System" -Name "DisableLogonBackgroundImage" -Value "1" -Type DWord -Force
    Write-Host "La interfaz de usuario ahora resuena con el cosmos." -ForegroundColor Green
}

# --- PASO 6: REGISTRO - RENDIMIENTO (RENDIMIENTO ABSOLUTO) ---
Function Step6-Registry-Absolute-Performance {
    Write-Host "`n`n[PASO 6/11] - Desatando el Rendimiento Absoluto del Registro..." -ForegroundColor 'Yellow'
    # Prioridad de CPU para aplicaciones en primer plano
    Set-ItemProperty -Path "HKLM:\SYSTEM\CurrentControlSet\Control\PriorityControl" -Name "Win32PrioritySeparation" -Value "26" -Type DWord -Force
    Write-Host "El rendimiento ha trascendido sus límites anteriores." -ForegroundColor Green
}

# --- PASO 7: OPTIMIZACIÓN SSD (IMPULSO DE ESTADO SÓLIDO) ---
Function Step7-Solid-State-Optimization {
    Write-Host "`n`n[PASO 7/11] - Activando el Impulso de Estado Sólido..." -ForegroundColor 'Yellow'
    powercfg -setactive 8c5e7fda-e8bf-4a96-9a85-a6e23a8c635c # Máximo rendimiento
    Write-Host "Las unidades de estado sólido ahora operan a máxima eficiencia." -ForegroundColor Green
}

# --- PASO 8: PRIVACIDAD (ESCUDO DE PRIVACIDAD INQUEBRANTABLE) ---
Function Step8-Unbreakable-Privacy {
    Write-Host "`n`n[PASO 8/11] - Erigiendo el Escudo de Privacidad Inquebrantable..." -ForegroundColor 'Yellow'
    # Desactivar telemetría
    Set-ItemProperty -Path "HKLM:\SOFTWARE\Policies\Microsoft\Windows\DataCollection" -Name "AllowTelemetry" -Value "0" -Type DWord -Force
    Write-Host "Tu privacidad es ahora una fortaleza impenetrable." -ForegroundColor Green
}

# --- PASO 9: SERVICIOS (AJUSTE DE SERVICIOS DEFINITIVO) ---
Function Step9-Ultimate-Service-Tuning {
    Write-Host "`n`n[PASO 9/11] - Realizando el Ajuste de Servicios Definitivo..." -ForegroundColor 'Yellow'
    # Desactivar servicios innecesarios (ejemplo: Fax)
    Set-Service -Name "Fax" -StartupType Disabled -Status Stopped
    Write-Host "Los servicios del sistema han sido afinados para la perfección." -ForegroundColor Green
}

# --- PASO 10: DEBLOAT (CONTROL DE LANZAMIENTO) ---
Function Step10-Launch-Control-Debloat {
    Write-Host "`n`n[PASO 10/11] - Iniciando Control de Lanzamiento y Debloat..." -ForegroundColor 'Yellow'
    # Eliminar aplicaciones preinstaladas (ejemplo: 3D Viewer)
    Get-AppxPackage *3dviewer* | Remove-AppxPackage
    Write-Host "El sistema ha sido purgado de software innecesario." -ForegroundColor Green
}
